<script>
document.addEventListener('DOMContentLoaded', function() {
    // Make sure the container is empty before initializing
    var calendarEl = document.getElementById('project-calendar');
    
    if (calendarEl && !calendarEl.classList.contains('fc')) {
        var calendar = new FullCalendar.Calendar(calendarEl, {
            initialView: 'dayGridMonth',
            locale: 'fr',
            firstDay: 1, // Monday first day of week
            headerToolbar: false, // Hide default toolbar completely
            height: 'auto',
            contentHeight: 'auto',
            aspectRatio: 1.5,
            events: [
                // Example events - replace with your dynamic data
                {
                    title: 'Réunion équipe',
                    start: '2026-04-15',
                    color: '#0d6efd'
                },
                {
                    title: 'Deadline projet',
                    start: '2026-04-25',
                    end: '2026-04-26',
                    color: '#dc3545'
                },
                {
                    title: 'Formation',
                    start: '2026-04-20T10:00:00',
                    color: '#198754'
                }
            ],
            eventClick: function(info) {
                // Handle event click
                alert('Événement: ' + info.event.title);
                // You can open a modal with event details here
                console.log('Event clicked:', info.event);
            },
            dateClick: function(info) {
                // Handle date click - add new event
                console.log('Date clicked: ' + info.dateStr);
                // You can open a modal to add event here
                alert('Ajouter un événement pour le ' + info.dateStr);
            },
            datesSet: function(info) {
                // Update custom toolbar when calendar changes view/date
                updateMonthYear(calendar);
                updateEventCount(calendar);
            }
        });
        
        calendar.render();
        
        // Update month/year display
        function updateMonthYear(cal) {
            const currentDate = cal.getDate();
            const monthNames = ['janvier', 'février', 'mars', 'avril', 'mai', 'juin', 
                               'juillet', 'août', 'septembre', 'octobre', 'novembre', 'décembre'];
            const monthYear = `${monthNames[currentDate.getMonth()]} ${currentDate.getFullYear()}`;
            const monthYearEl = document.getElementById('calendar-month-year');
            if (monthYearEl) monthYearEl.textContent = monthYear;
        }
        
        // Update event count
        function updateEventCount(cal) {
            const events = cal.getEvents();
            const eventCountEl = document.getElementById('event-count');
            if (eventCountEl) eventCountEl.textContent = events.length;
        }
        
        // Custom toolbar controls
        document.getElementById('calendar-today')?.addEventListener('click', function() {
            calendar.today();
        });
        
        document.getElementById('calendar-prev')?.addEventListener('click', function() {
            calendar.prev();
        });
        
        document.getElementById('calendar-next')?.addEventListener('click', function() {
            calendar.next();
        });
        
        // View switching
        document.querySelectorAll('[data-view]').forEach(btn => {
            btn.addEventListener('click', function() {
                const view = this.getAttribute('data-view');
                calendar.changeView(view);
                
                // Update active state
                document.querySelectorAll('[data-view]').forEach(b => {
                    b.classList.remove('active', 'btn-primary');
                    b.classList.add('btn-outline-primary');
                });
                this.classList.remove('btn-outline-primary');
                this.classList.add('active', 'btn-primary');
            });
        });
        
        // Handle collapse/expand to resize calendar
        const calendarBody = document.getElementById('calendarBody');
        const calendarArrow = document.getElementById('calendarArrow');
        
        if (calendarBody) {
            calendarBody.addEventListener('shown.bs.collapse', function() {
                calendar.updateSize();
                if (calendarArrow) calendarArrow.style.transform = 'rotate(180deg)';
            });
            
            calendarBody.addEventListener('hidden.bs.collapse', function() {
                if (calendarArrow) calendarArrow.style.transform = 'rotate(0deg)';
            });
        }
        
        // Initial update
        updateMonthYear(calendar);
        updateEventCount(calendar);
        
        // Store calendar globally for debugging
        window.projectCalendar = calendar;
    } else {
        console.error('Calendar container not found or already initialized');
    }
});
</script>
